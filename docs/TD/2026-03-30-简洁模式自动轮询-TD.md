**改动**：在两个 `showToast('邮箱地址已复制', 'success')` 之后各插入 2 行：


**改动后完整函数**：
```javascript
async function copyEmail(email) {
    try {
        if (navigator.clipboard && navigator.clipboard.writeText && window.isSecureContext) {
            await navigator.clipboard.writeText(email);
            showToast('邮箱地址已复制', 'success');
            window.dispatchEvent(new CustomEvent('email-copied', { detail: { email } }));
            return true;
        }
        const textarea = document.createElement('textarea');
        textarea.value = email;
        textarea.setAttribute('readonly', 'readonly');
        textarea.style.position = 'fixed';
        textarea.style.top = '-9999px';
        textarea.style.left = '-9999px';
        document.body.appendChild(textarea);
        textarea.focus();
        textarea.select();
        const copied = document.execCommand('copy');
        document.body.removeChild(textarea);
        if (!copied) {
            throw new Error('document.execCommand(copy) returned false');
        }
        showToast('邮箱地址已复制', 'success');
        window.dispatchEvent(new CustomEvent('email-copied', { detail: { email } }));
        return true;
    } catch (error) {
        console.error('复制邮箱地址失败:', error);
        showToast('复制失败，请手动复制', 'error');
        return false;
    }
}
```

### 2.2 `static/js/main.js` — showToast 新增 duration 参数

**改动位置**：`showToast()` 函数签名（第 1124 行）

**改动**：新增第四个参数 `duration`，默认 `null`（保持原有行为）：

```javascript
function showToast(message, type = 'info', errorDetail = null, duration = null) {
    // ... 现有逻辑不变 ...

    const autoHideMs = duration !== null ? duration : ((errorDetail && type === 'error') ? 8000 : 3000);
    setTimeout(() => {
        toast.style.opacity = '0';
        toast.style.transform = 'translateX(30px)';
        setTimeout(() => toast.remove(), 300);
    }, autoHideMs);
}
```

**说明**：
- `duration = null` 时保持原有默认行为（success/info 3000ms，error with detail 8000ms）
- 轮询相关 Toast 传入 `5000`，实现 PRD 要求的 5 秒自动消失

### 2.3 `static/js/features/mailbox_compact.js` — 轮询引擎

**文件结构**：全局函数（非 IIFE），所有函数定义在全局作用域，通过 `<script>` 标签加载。

**新增位置**：文件末尾 `window.addEventListener('ui-language-changed', ...)` 之前。

#### 2.3.2 辅助函数

```javascript
// 通过邮箱地址查找账号对象
function getCompactAccountByEmail(email) {
    return getCompactVisibleAccounts().find(acc => acc.email === email) || null;
}

// 查找账号行 DOM 元素
// 注意：实际行容器类名是 .mail-row（renderCompactAccountList 第 240 行），不是 .compact-account-row
// 注意：不使用 CSS.escape，因为 [onclick*="..."] 是 HTML 属性子串匹配，不是 CSS 选择器语义
//        邮箱地址通常不含 " 和 \ 字符，直接使用即可
function findCompactAccountRow(email) {
    const btn = document.querySelector(`.pull-button[onclick*="${email}"]`);
    return btn ? btn.closest('.mail-row') : null;
}

// 查找或创建邮箱行内的脉冲圆点
function ensurePollDot(rowElement) {
    let dot = rowElement.querySelector('.compact-poll-dot');
    if (!dot) {
        dot = document.createElement('span');
        dot.className = 'compact-poll-dot';
        const mailCard = rowElement.querySelector('.mail-card');
        if (mailCard) {
            const button = mailCard.querySelector('.mail-card-button');
            if (button) {
                button.style.position = 'relative';
                button.appendChild(dot);
            }
        }
    }
    return dot;
}

// 更新单个邮箱行的轮询 UI
function updateCompactPollUI(email, state, remaining) {
    const row = findCompactAccountRow(email);
    if (!row) return;

    const pullBtn = row.querySelector('.pull-button');
    if (!pullBtn) return;

    if (state === 'polling') {
        pullBtn.textContent = translateCompactText('停止监听') + ` (${remaining}s)`;
        pullBtn.classList.add('compact-poll-active');
        pullBtn.setAttribute('data-poll-email', email);
        pullBtn.onclick = function(e) {
            e.stopPropagation();
            stopCompactAutoPoll(email);
        };
        ensurePollDot(row);
    } else {
        pullBtn.textContent = translateCompactText('拉取');
        pullBtn.classList.remove('compact-poll-active');
        pullBtn.removeAttribute('data-poll-email');
        // 恢复原始 onclick：从 compactPollMap（已删除）中无法获取 accountId，
        // 但 updateCompactPollUI('stopped') 总是在 stopCompactAutoPoll 内部调用，
        // 此时 Map 中已无该条目。因此从全局 accountsCache 反查 accountId
        pullBtn.onclick = function() {
            const acc = getCompactAccountByEmail(email);
            if (acc) {
                refreshCompactAccount(acc.id, this);
            }
        };
        const dot = row.querySelector('.compact-poll-dot');
        if (dot) dot.remove();
    }
}
```

**关键修正**：
- `.compact-account-row` → `.mail-row`（与 renderCompactAccountList 第 240 行一致）
- `CSS.escape(email)` → 直接用 `email`（`[onclick*="..."]` 是属性子串匹配，不是 CSS 选择器语义）
- 恢复按钮 onclick 时从 `accountsCache` 反查 `accountId`，而非依赖 DOM 属性（实际 HTML 无 `data-account-id`）

**注意**：`renderCompactAccountList()` 会全量重建 DOM，因此需要在全量渲染后调用 `reapplyAllCompactPollUI()` 重新应用轮询状态。

#### 2.3.3 核心轮询函数

```javascript
// 启动自动轮询
async function startCompactAutoPoll(email, accountId) {
    // 如果该邮箱已在轮询，先停止
    if (compactPollMap.has(email)) {
        stopCompactAutoPoll(email, true); // silent
    }

    console.log(`[简洁轮询] 启动: ${email}, 间隔: ${compactPollInterval}s, 时长: ${compactPollMaxDuration}s`);

    // ① 先更新 UI（在 baseline 拉取之前，避免慢网络下用户看到"无反应"）
    //    同时创建 PollState 并加入 Map，防止重入
    const state = {
        timer: null,
        baselineIds: new Set(),
        accountId: accountId,
        errorCount: 0,
        startTime: Date.now(),
        isPolling: false,
        uiMissCount: 0    // 连续 UI 更新失败计数（findCompactAccountRow 返回 null）
    };
    compactPollMap.set(email, state);
    updateCompactPollUI(email, 'polling', compactPollMaxDuration);
    startGlobalCountdown();

    // ② 拉取邮件建立 baseline
    try {
        const [inboxRes, junkRes] = await Promise.allSettled([
            fetch(`/api/emails/${encodeURIComponent(email)}?folder=inbox&skip=0&top=3`),
            fetch(`/api/emails/${encodeURIComponent(email)}?folder=junkemail&skip=0&top=3`)
        ]);
        for (const result of [inboxRes, junkRes]) {
            if (result.status === 'fulfilled' && result.value.ok) {
                const payload = await result.value.json();
                if (payload.success && Array.isArray(payload.emails)) {
                    payload.emails.forEach(e => state.baselineIds.add(e.id));
                }
            }
        }
    } catch (e) {
        console.warn('[简洁轮询] 建立 baseline 失败:', e);
    }

    // ③ 立即执行一次轮询
    pollSingleEmail(email);

    // ④ 设置定时器
    state.timer = setInterval(() => pollSingleEmail(email), compactPollInterval * 1000);
}

// 执行单次轮询
async function pollSingleEmail(email) {
    const state = compactPollMap.get(email);
    if (!state) return;

    // 防重入锁
    if (state.isPolling) return;
    state.isPolling = true;

    try {
        // 检查超时
        const elapsed = (Date.now() - state.startTime) / 1000;
        if (elapsed >= compactPollMaxDuration) {
            stopCompactAutoPoll(email);
            showToast(`${email}: ${translateCompactText('监听超时，未检测到新邮件')}`, 'info', null, COMPACT_POLL_TOAST_DURATION);
            return;
        }

        // 并行拉取 inbox + junkemail
        const [inboxRes, junkRes] = await Promise.allSettled([
            fetch(`/api/emails/${encodeURIComponent(email)}?folder=inbox&skip=0&top=3`),
            fetch(`/api/emails/${encodeURIComponent(email)}?folder=junkemail&skip=0&top=3`)
        ]);

        // 检查 404（账号被删除）
        for (const result of [inboxRes, junkRes]) {
            if (result.status === 'fulfilled' && result.value && result.value.status === 404) {
                stopCompactAutoPoll(email);
                showToast(`${email}: ${translateCompactText('账号已被删除，已停止监听')}`, 'error', null, COMPACT_POLL_TOAST_DURATION);
                return;
            }
        }

        // 检查请求是否成功
        let hasSuccess = false;
        let foundNewEmail = false;
        let accountSummary = null;

        for (const result of [inboxRes, junkRes]) {
            if (result.status !== 'fulfilled' || !result.value.ok) continue;
            try {
                const payload = await result.value.json();
                if (!payload.success || !Array.isArray(payload.emails)) continue;
                hasSuccess = true;
                if (payload.account_summary) {
                    accountSummary = payload.account_summary;
                }
                // 对比 baseline
                const newEmails = payload.emails.filter(e => !state.baselineIds.has(e.id));
                if (newEmails.length > 0) {
                    foundNewEmail = true;
                }
            } catch (e) {
                // JSON 解析失败
            }
        }

        if (!hasSuccess) {
            state.errorCount++;
            if (state.errorCount >= 3) {
                stopCompactAutoPoll(email);
                showToast(`${email}: ${translateCompactText('拉取失败，已停止监听')}`, 'error', null, COMPACT_POLL_TOAST_DURATION);
            }
            return;
        }

        // 成功请求后重置错误计数
        state.errorCount = 0;

        if (foundNewEmail) {
            // 发现新邮件
            // ① 更新摘要缓存
            if (accountSummary && typeof syncAccountSummaryToAccountCache === 'function') {
                syncAccountSummaryToAccountCache(email, accountSummary);
            }

            // ② 局部更新 UI（不调用 renderCompactAccountList 全量重渲染）
            //    更新该行的验证码、邮件摘要等文字
            updateSingleRowFromCache(email);

            // ③ 自动提取验证码（区分网络错误和无验证码）
            try {
                const extractRes = await fetch(`/api/emails/${encodeURIComponent(email)}/extract-verification`);
                if (!extractRes.ok) {
                    // 接口异常（网络/服务端错误），仅 log 警告
                    console.warn(`[简洁轮询] 验证码提取接口异常: ${extractRes.status}`);
                }
                const extractData = await extractRes.json();
                if (extractData.success && extractData.data && extractData.data.verification_code) {
                    const code = extractData.data.verification_code;
                    if (typeof copyToClipboard === 'function') {
                        await copyToClipboard(code);
                    }
                    showToast(`${email}: ${translateCompactText('检测到验证码')}: ${code}，${translateCompactText('已复制到剪贴板')}`, 'success', null, COMPACT_POLL_TOAST_DURATION);
                } else {
                    showToast(`${email}: ${translateCompactText('发现新邮件')}`, 'success', null, COMPACT_POLL_TOAST_DURATION);
                }
            } catch (e) {
                console.warn(`[简洁轮询] 验证码提取失败:`, e);
                showToast(`${email}: ${translateCompactText('发现新邮件')}`, 'success', null, COMPACT_POLL_TOAST_DURATION);
            }

            // ④ 停止轮询
            stopCompactAutoPoll(email);
        }
        // 无新邮件 → 继续下一次轮询（由 setInterval 自动触发）

    } catch (error) {
        console.error(`[简洁轮询] ${email} 轮询失败:`, error);
        // 只有 fetch 网络异常才会到此处，递增 errorCount
        state.errorCount++;
        if (state.errorCount >= 3) {
            stopCompactAutoPoll(email);
            showToast(`${email}: ${translateCompactText('拉取失败，已停止监听')}`, 'error', null, COMPACT_POLL_TOAST_DURATION);
        }
    } finally {
        // 仅当 Map 中仍存在该条目时才重置锁（避免操作已删除的对象）
        if (compactPollMap.has(email)) {
            state.isPolling = false;
        }
    }
}

// 停止指定邮箱的轮询
function stopCompactAutoPoll(email, silent = false) {
    const state = compactPollMap.get(email);
    if (!state) return;

    if (state.timer) {
        clearInterval(state.timer);
    }
    compactPollMap.delete(email);

    // 更新 UI
    updateCompactPollUI(email, 'stopped');

    // 如果没有其他轮询了，停止全局倒计时
    if (compactPollMap.size === 0 && compactPollCountdownTimer) {
        clearInterval(compactPollCountdownTimer);
        compactPollCountdownTimer = null;
    }

    if (!silent) {
        showToast(`${email}: ${translateCompactText('已停止监听')}`, 'info', null, COMPACT_POLL_TOAST_DURATION);
    }
}

// 停止所有轮询
function stopAllCompactAutoPolls() {
    for (const email of compactPollMap.keys()) {
        stopCompactAutoPoll(email, true);
    }
}
```

**关键修正**：
- **baseline 拉取前先更新 UI**：`compactPollMap.set()` 和 `updateCompactPollUI()` 移到 baseline 拉取之前，避免慢网络下"无反应"
- **删除冗余步骤③**：发现新邮件后不再重新应用轮询指示器（紧接着就 stopCompactAutoPoll 会清除）
- **finally 中检查 Map 存在性**：`if (compactPollMap.has(email))` 避免操作已删除的对象
- **验证码提取区分网络错误和无验证码**：`!extractRes.ok` 时仅 log 警告，不影响后续逻辑
- **所有轮询 Toast 传入 5000ms**：通过 `COMPACT_POLL_TOAST_DURATION` 常量统一

#### 2.3.4 全局倒计时

```javascript
function startGlobalCountdown() {
    if (compactPollCountdownTimer) return; // 已启动

    compactPollCountdownTimer = setInterval(() => {
        const now = Date.now();
        const timedOut = []; // 收集超时的邮箱，循环结束后批量处理

        for (const [email, state] of compactPollMap) {
            const elapsed = (now - state.startTime) / 1000;
            const remaining = Math.max(0, Math.ceil(compactPollMaxDuration - elapsed));
            updateCompactPollUI(email, 'polling', remaining);

            if (elapsed >= compactPollMaxDuration) {
                timedOut.push(email);
            }
        }

        // 批量停止超时的邮箱，避免在 for...of 中修改 Map 导致布局抖动
        for (const email of timedOut) {
            stopCompactAutoPoll(email);
            showToast(`${email}: ${translateCompactText('监听超时，未检测到新邮件')}`, 'info', null, COMPACT_POLL_TOAST_DURATION);
        }

        // 所有轮询已停止时，清理定时器
        if (compactPollMap.size === 0) {
            clearInterval(compactPollCountdownTimer);
            compactPollCountdownTimer = null;
        }
    }, 1000);
}
```

**修正**：超时邮箱收集到数组中批量处理，避免在 for...of 迭代 Map 时连续触发 DOM 更新造成布局抖动。

#### 2.3.5 局部行更新

```javascript
// 从 accountsCache 更新单行的 DOM（避免全量重渲染）
// meta 拼接逻辑与 renderCompactAccountList 第 233-237 行保持一致：from + folder + received_at
function updateSingleRowFromCache(email) {
    const account = getCompactAccountByEmail(email);
    const row = findCompactAccountRow(email);
    if (!account || !row) {
        // 兜底：如果连续多次找不到行，可能是 DOM 结构变化
        const state = compactPollMap.get(email);
        if (state) {
            state.uiMissCount = (state.uiMissCount || 0) + 1;
            if (state.uiMissCount >= 5) {
                console.warn(`[简洁轮询] ${email} 连续 ${state.uiMissCount} 次 UI 更新失败，停止轮询`);
                stopCompactAutoPoll(email);
                showToast(`${email}: ${translateCompactText('页面元素丢失，已停止监听')}`, 'error', null, COMPACT_POLL_TOAST_DURATION);
            }
        }
        return;
    }

    // 找到行后重置计数器
    const state = compactPollMap.get(email);
    if (state) state.uiMissCount = 0;

    // 更新验证码
    const codeBtn = row.querySelector('.code-button');
    if (codeBtn) {
        const code = account.latest_verification_code || '';
        codeBtn.textContent = code || translateCompactText('暂无');
        codeBtn.classList.toggle('empty', !code);
    }

    // 更新邮件摘要
    const subjectEl = row.querySelector('.snippet-subject');
    if (subjectEl) {
        subjectEl.textContent = account.latest_email_subject || translateCompactText('暂无邮件');
    }

    const metaEl = row.querySelector('.snippet-meta');
    if (metaEl) {
        // 与 renderCompactAccountList 第 233-237 行保持一致：from + folder + received_at
        const parts = [
            account.latest_email_from || '',
            account.latest_email_folder || '',
            account.latest_email_received_at || ''
        ].filter(Boolean);
        metaEl.textContent = parts.join(' · ') || translateCompactText('暂无邮件摘要');
    }
}
```

**关键修正**：
- **snippet-meta 格式一致性**：加上 `latest_email_folder`，与 `renderCompactAccountList` 第 233-237 行的 `latestEmailMeta` 拼接逻辑一致
- **UI 更新失败兜底**：连续 5 次 `findCompactAccountRow` 返回 null 时停止轮询并提示

#### 2.3.6 全量渲染后重新应用轮询状态

```javascript
// 在 renderCompactAccountList() 被调用后，重新应用所有轮询邮箱的 UI 状态
function reapplyAllCompactPollUI() {
    if (compactPollMap.size === 0) return; // early return，无轮询时跳过

    const now = Date.now();
    for (const [email, state] of compactPollMap) {
        // 全量渲染后 DOM 元素已替换，重置 UI 失败计数器
        state.uiMissCount = 0;
        const remaining = Math.max(0, Math.ceil(compactPollMaxDuration - (now - state.startTime) / 1000));
        updateCompactPollUI(email, 'polling', remaining);
    }
}
```

**集成点**：需要在 `renderCompactAccountList()` 函数末尾调用 `reapplyAllCompactPollUI()`。

**修正**：入口加 `if (compactPollMap.size === 0) return;` 避免无轮询时不必要的循环；全量渲染后重置 `uiMissCount`。

#### 2.3.7 设置同步

```javascript
// 由 main.js 在设置加载和设置保存后调用，将设置值同步到本文件的全局变量
function applyCompactPollSettings(settings) {
    compactPollEnabled = !!settings.enabled;
    compactPollInterval = parseInt(settings.interval) || 10;
    compactPollMaxDuration = parseInt(settings.maxDuration) || 60;
    applyCompactPollSettingsToRunningPolls();
}

// 应用设置变更到正在运行的轮询
function applyCompactPollSettingsToRunningPolls() {
    if (!compactPollEnabled) {
        stopAllCompactAutoPolls();
        return;
    }

    // 重建所有正在轮询邮箱的定时器（应用新间隔）
    for (const [email, state] of compactPollMap) {
        if (state.timer) {
            clearInterval(state.timer);
            state.timer = setInterval(() => pollSingleEmail(email), compactPollInterval * 1000);
        }
    }
}
```

**说明**：`main.js` 和 `mailbox_compact.js` 各有一份设置变量。`main.js` 负责从后端加载和保存，通过 `applyCompactPollSettings()` 同步到 `mailbox_compact.js` 的全局变量。`maxDuration` 不需要重建定时器，因为 `pollSingleEmail` 和 `startGlobalCountdown` 中动态读取全局变量，修改后立即生效。

#### 2.3.8 事件监听与可见性检测

```javascript
// 监听 email-copied 事件
window.addEventListener('email-copied', function(e) {
    const { email } = e.detail;
    if (!compactPollEnabled) return;
    if (typeof mailboxViewMode === 'undefined' || mailboxViewMode !== 'compact') return;
    if (typeof isTempEmailGroup !== 'undefined' && isTempEmailGroup) return;

    const account = getCompactAccountByEmail(email);
    if (!account) return;

    startCompactAutoPoll(email, account.id);
});

// 页面可见性检测
document.addEventListener('visibilitychange', function() {
    if (document.hidden) {
        // 暂停所有轮询定时器（保留 Map 状态）
        for (const [email, state] of compactPollMap) {
            if (state.timer) {
                clearInterval(state.timer);
                state.timer = null;
            }
        }
        // 暂停全局倒计时
        if (compactPollCountdownTimer) {
            clearInterval(compactPollCountdownTimer);
            compactPollCountdownTimer = null;
        }
    } else {
        // 恢复所有轮询
        const timedOut = [];
        for (const [email, state] of compactPollMap) {
            // 检查是否已超时
            const elapsed = (Date.now() - state.startTime) / 1000;
            if (elapsed >= compactPollMaxDuration) {
                timedOut.push(email);
                continue;
            }
            // 立即执行一次
            pollSingleEmail(email);
            // 恢复定时器
            state.timer = setInterval(() => pollSingleEmail(email), compactPollInterval * 1000);
        }
        // 批量处理超时的邮箱
        for (const email of timedOut) {
            stopCompactAutoPoll(email);
            showToast(`${email}: ${translateCompactText('监听超时，未检测到新邮件')}`, 'info', null, COMPACT_POLL_TOAST_DURATION);
        }
        // 恢复全局倒计时
        if (compactPollMap.size > 0) {
            startGlobalCountdown();
        }
    }
});
```

**设计决策**：恢复轮询时不额外检查 `mailboxViewMode`。原因：简洁模式轮询与标准模式轮询互不影响（PRD 4.9），即使切回前台时不在 compact 模式，轮询在后台静默运行不会有副作用——只是用户看不到指示器。下次切回 compact 模式时 `reapplyAllCompactPollUI()` 会恢复指示器。

## 5. 事件流时序

```
[用户操作] 点击邮箱名 → copyEmail(email)
                    ↓
         navigator.clipboard.writeText(email)
                    ↓
         showToast('邮箱地址已复制', 'success')
                    ↓
         dispatchEvent('email-copied', { email })  ← 2 行新增
                    ↓
[事件监听] mailbox_compact.js 'email-copied' listener
                    ↓
         检查: compactPollEnabled && mailboxViewMode === 'compact' && !isTempEmailGroup
                    ↓
         getCompactAccountByEmail(email) → accountId
                    ↓
         startCompactAutoPoll(email, accountId)
                    ↓
    ① compactPollMap.set(email, state)  ← 先创建状态
    ② updateCompactPollUI(email, 'polling', maxDuration)  ← 立即显示指示器
    ③ Promise.allSettled([inbox, junkemail])  ← 拉取 baseline
    ④ pollSingleEmail(email)  ← 立即首次轮询
    ⑤ setInterval(pollSingleEmail, interval)
                    ↓
[轮询循环] pollSingleEmail(email)
    防重入检查 → 超时检查 → fetch → 404检查 → 对比baseline
                    ↓
    ┌─────── 无新邮件 ──────┐    ┌─────── 有新邮件 ──────┐
    │ errorCount=0          │    │ syncAccountSummary()   │
    │ 等 setInterval 触发    │    │ updateSingleRowFromCache() │
    └───────────────────────┘    │ extractVerification()   │
                                 │ copyToClipboard(code)   │
                                 │ showToast(5s)           │
                                 │ stopCompactAutoPoll()   │
                                 └───────────────────────┘
```
